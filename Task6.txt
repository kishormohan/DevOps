1.While Loop:

Syntax:

while <condition>
do
    <command 1>
    <command 2>
    <etc>
done

_______________________________________________________________________________________________________________________________________________________________________________________________
2.For Loop:

Syntax

for <var> in <value_1 value_2 ... value_n>
do
    <command 1>
    <command 2>
    <etc>
done
_______________________________________________________________________________________________________________________________________________________________________________________________


3.Until Loop:

Syntax

until <condition>
do
    <command 1>
    <command 2>
    <etc>
done

_______________________________________________________________________________________________________________________________________________________________________________________________

4.Break

for i in {1..5}; do
echo $i
if [ $i -eq 3 ]; then
break
fi
done
echo “For loop terminated”

_______________________________________________________________________________________________________________________________________________________________________________________________

5.Continue

for i in `seq 1 10`
do
   if (( $i==5))
       then
       continue
   fi
   echo $i
done
